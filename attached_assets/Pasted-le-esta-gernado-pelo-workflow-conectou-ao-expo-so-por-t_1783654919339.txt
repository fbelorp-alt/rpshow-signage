 * 1. REMOVIDO replayAsync() — causa race condition no Android

 * 3. Um único <Video> com key={currentIndex} — remonta limpo a cada troca
 * 4. Para o vídeo ANTES de avançar
 * 5. setTimeout(50) para sair do callback nativo do expo-av
 * 6. endedRef + genRef evitam callbacks duplicados/antigos

// SUBSTITUA o VideoPlayer e a lógica de advance por isto:
// ============================================================

import { useState, useEffect, useRef, useCallback } from "react";
import { View, StyleSheet } from "react-native";
import { Video, ResizeMode, AVPlaybackStatus } from "expo-av";

// --- refs essenciais ---
const videoRef = useRef<Video>(null);
const endedRef = useRef(false);
const genRef = useRef(0);
const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
const itemsLenRef = useRef(0);

// --- advance limpo ---
const advance = useCallback(() => {
  if (timerRef.current) {
    clearTimeout(timerRef.current);
    timerRef.current = null;
  }
  genRef.current += 1;
  endedRef.current = false;

  setCurrentIndex((prev) => {
    const len = itemsLenRef.current;
    if (len <= 0) return 0;
    // Funciona igual para o último item: (N-1 + 1) % N = 0
    return (prev + 1) % len;
  });
}, []);

// --- finishCurrent: chamado quando o vídeo termina ---
const finishCurrent = useCallback(() => {
  // Guard: só executa uma vez por item
  if (endedRef.current) return;
  endedRef.current = true;

  if (timerRef.current) {
    clearTimeout(timerRef.current);
    timerRef.current = null;
  }

  // CRÍTICO: para o MediaPlayer nativo ANTES de mudar o estado
  // Isso evita a race condition do Android reiniciar o vídeo
  videoRef.current
    ?.setStatusAsync({
      shouldPlay: false,
      isLooping: false,
      positionMillis: 0,
    })
    .catch(() => {});

  // CRÍTICO: sai do callback nativo do expo-av antes de setState
  // Sem isso, o Android pode reprocessar didJustFinish e reiniciar
  setTimeout(() => {
    advance();
  }, 50);
}, [advance]);

// --- fallback timer (caso didJustFinish não dispare) ---
useEffect(() => {
  if (!currentItem || currentItem.mediaType !== "video") return;
  endedRef.current = false;
  if (timerRef.current) clearTimeout(timerRef.current);

  const fallbackMs = ((currentItem.durationSeconds || 30) + 15) * 1000;
  timerRef.current = setTimeout(finishCurrent, fallbackMs);

  return () => {
    if (timerRef.current) clearTimeout(timerRef.current);
  };
}, [currentIndex, currentItem, finishCurrent]);

// ============================================================
// RENDER — UM ÚNICO VÍDEO (sem pool current/next)
// ============================================================

{currentItem?.mediaType === "video" && (
  <Video
    // KEY força remount limpo a cada troca de item
    // Isso elimina estado residual do MediaPlayer no Android
    key={`video-${currentIndex}-g${genRef.current}`}
    ref={videoRef}
    source={{ uri: currentItem.url /* ou slotUrl do cache */ }}
    style={StyleSheet.absoluteFill}
    resizeMode={ResizeMode.CONTAIN}
    shouldPlay={true}
    isLooping={false}          // NUNCA true

        finishCurrent();
      }
    }}
    onError={(e) => {
      console.log(`[VideoPlayer] ERROR idx=${currentIndex}`, e);
      finishCurrent(); // se falhar, avança
    }}
  />
)}
